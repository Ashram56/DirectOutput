102  private void LedControlIniFiles_MouseDown(object sender, MouseEventArgs e)
103  {
104  // Get the index of the item the mouse is below.
105  rowIndexFromMouseDown = LedControlIniFiles.HitTest(e.X, e.Y).RowIndex;
106  if (rowIndexFromMouseDown != -1)
107  {
108  // Remember the point where the mouse down occurred.
109  // The DragSize indicates the size that the mouse can move
110  // before a drag event should be started.
111  Size dragSize = SystemInformation.DragSize;
112 
113  // Create a rectangle using the DragSize, with the mouse position being
114  // at the center of the rectangle.
115  dragBoxFromMouseDown = new Rectangle(new Point(e.X - (dragSize.Width / 2),
116  e.Y - (dragSize.Height / 2)),
117  dragSize);
118  }
119  else
120  // Reset the rectangle if the mouse is not over an item in the ListBox.
121  dragBoxFromMouseDown = Rectangle.Empty;
122  }

129  private void LedControlIniFiles_DragDrop(object sender, DragEventArgs e)
130  {
131  // The mouse locations are relative to the screen, so they must be
132  // converted to client coordinates.
133  Point clientPoint = LedControlIniFiles.PointToClient(new Point(e.X, e.Y));
134 
135  // Get the row index of the item the mouse is below.
136  rowIndexOfItemUnderMouseToDrop =
137  LedControlIniFiles.HitTest(clientPoint.X, clientPoint.Y).RowIndex;
138 
139  // If the drag operation was a move then remove and insert the row.
140  if (e.Effect == DragDropEffects.Move)
141  {
142  DataGridViewRow rowToMove = e.Data.GetData(
143  typeof(DataGridViewRow)) as DataGridViewRow;
144 
145  LedControlIniFileList L = new LedControlIniFileList();
146  int Cnt = 0;
147  int Number = 1;
148  foreach (LedControlIniFile I in GlobalConfig.LedControlIniFiles)
149  {
150  if (Cnt != rowIndexFromMouseDown)
151  {
152  I.LedWizNumber = Number;
153  L.Add(I);
154  Number++;
155  }
156  if (Cnt == rowIndexOfItemUnderMouseToDrop)
157  {
158  LedControlIniFile I1 = GlobalConfig.LedControlIniFiles[rowIndexFromMouseDown];
159  I1.LedWizNumber = Number;
160  L.Add(I1);
161  }
162  Cnt++;
163  }
164  GlobalConfig.LedControlIniFiles = L;
165  this.LedControlIniFiles.DataSource = L;
166  this.LedControlIniFiles.Refresh();
167 
168  Modified = true;
169 
170  }
